import java.util.*;

public class Multiply
{

public static void main(String[] args)

	{

	    Scanner s=new Scanner(System.in);

	    int i,n;

	    System.out.println("enter n values");

	    n=s.nextInt();

	    for(i=1;i<=10;i++)

	    {

		System.out.println(i+"*"+n+"="+i*n);

	    }

	}

}
